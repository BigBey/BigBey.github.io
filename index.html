 public static Triplet<Integer, Integer, Integer> findMaximumSubarray(int[] array, int low, int high) {
        int sum, maxLeft, maxRight, sumOfSubarray;
        sum = array[low];
        sumOfSubarray = array[low];
        maxLeft = low;
        maxRight = low;
        for (int j = low; j <= high - 1; j++) {
            if (array[j + 1] > 0) {
                if (sum < 0 && sumOfSubarray < array[j + 1]) {
                    sum = array[j + 1];
                    sumOfSubarray = array[j + 1];
                    maxLeft = j + 1;
                    maxRight = j + 1;
                } else if (sum + array[j + 1] > sumOfSubarray) {
                    sum = sum + array[j + 1];
                    sumOfSubarray = sum;
                    maxRight = j + 1;
                }else{
                    sum = sum + array[j + 1];
                }
            } else {
                sum = sum + array[j + 1];
            }
        }
        return new Triplet<Integer, Integer, Integer>(maxLeft, maxRight, sumOfSubarray);
    }
